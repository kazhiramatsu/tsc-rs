```rust
use crate::kind::SyntaxKind;

bitflags::bitflags! {
    #[derive(Clone, Copy, Default)]
    pub struct TokenFlags: u32 {} // members generated by M0 codegen — do not hand-write
}

pub struct Scanner<'t> {
    /// WTF-8: JS source may contain lone surrogates. All `pos` values
    /// are BYTE offsets; the per-file byte↔UTF-16 map (built once in
    /// the harness) converts at the diagnostic boundary only.
    text: &'t [u8],
    end: usize,
    /// current scan position (== end of current token after scan())
    pub pos: usize,
    /// start of current token INCLUDING leading trivia
    pub full_start_pos: usize,
    /// start of current token EXCLUDING trivia
    pub token_start: usize,
    pub token: SyntaxKind,
    pub token_value: String,
    pub token_flags: TokenFlags,
    /// JSX vs Standard — set per file extension
    pub language_variant: LanguageVariant,
    /// diagnostics sink: the PARSER owns dedup (impl-parser §1);
    /// the scanner reports every error it finds
    pub on_error: fn(&mut Vec<ScanError>, ScanError),
    pub errors: Vec<ScanError>,
}

pub struct ScanError {
    pub message: &'static crate::diags::DiagnosticMessage,
    pub start: usize,     // byte offset
    pub length: usize,
    pub args: Vec<String>,
}

#[derive(Clone, Copy, PartialEq)]
pub enum LanguageVariant { Standard, Jsx }

/// full scanner state — save/restore for speculation (stage 1.7)
#[derive(Clone)]
pub struct ScannerState {
    pos: usize, full_start_pos: usize, token_start: usize,
    token: SyntaxKind, token_value: String, token_flags: TokenFlags,
    error_len: usize,
}

impl<'t> Scanner<'t> {
    pub fn save(&self) -> ScannerState {
        ScannerState { pos: self.pos, full_start_pos: self.full_start_pos,
            token_start: self.token_start, token: self.token,
            token_value: self.token_value.clone(), token_flags: self.token_flags,
            error_len: self.errors.len() }
    }
    pub fn restore(&mut self, s: ScannerState) {
        self.pos = s.pos; self.full_start_pos = s.full_start_pos;
        self.token_start = s.token_start; self.token = s.token;
        self.token_value = s.token_value; self.token_flags = s.token_flags;
        self.errors.truncate(s.error_len);
    }
    #[inline] fn ch(&self, at: usize) -> i32 {
        if at >= self.end { -1 } else { self.code_point_at(at) } // WTF-8 decode
    }
}
```

```rust
impl<'t> Scanner<'t> {
    pub fn scan(&mut self) -> SyntaxKind {
        self.full_start_pos = self.pos;
        self.token_flags = TokenFlags::empty();
        loop {
            self.token_start = self.pos;
            if self.pos >= self.end { self.token = SyntaxKind::EndOfFileToken; return self.token; }
            let ch = self.ch(self.pos);
            // shebang: only at position 0 (tsc scan() head)
            if ch == c::HASH && self.pos == 0 && self.ch(1) == c::BANG {
                todo_port!("scanShebangTrivia — skip to line end, continue");
            }
            match ch {
                c::LINE_FEED | c::CARRIAGE_RETURN => {
                    self.token_flags |= TokenFlags::PRECEDING_LINE_BREAK;
                    self.pos += 1; continue;    // trivia (skipTrivia=true model)
                }
                c::TAB | c::VERTICAL_TAB | c::FORM_FEED | c::SPACE => { self.pos += 1; continue; }
                c::SLASH => {
                    if self.ch(self.pos + 1) == c::SLASH {   // line comment
                        todo_port!("single-line comment skip; sets no token");
                    }
                    if self.ch(self.pos + 1) == c::ASTERISK { // block comment
                        todo_port!("multi-line comment skip; unterminated => error 1010 at token_start; PrecedingLineBreak if it contains one");
                    }
                    if self.ch(self.pos + 1) == c::EQUALS {
                        self.pos += 2; self.token = SyntaxKind::SlashEqualsToken; return self.token;
                    }
                    self.pos += 1; self.token = SyntaxKind::SlashToken; return self.token;
                }
                c::DOUBLE_QUOTE | c::SINGLE_QUOTE => {
                    self.token_value = self.scan_string(/*jsx_attr*/ false);
                    self.token = SyntaxKind::StringLiteral; return self.token;
                }
                c::BACKTICK => { self.token = self.scan_template_and_set_token_value(false); return self.token; }
                c::_0..=c::_9 => { self.token = self.scan_number(); return self.token; }
                // ---- punctuation: one arm per tsc case, IN ORDER ----
                // NOTE: `>` is ALWAYS a single GreaterThanToken here;
                // >>,>>>,>=,>>=,>>>= exist ONLY in re_scan_greater_token.
                c::OPEN_BRACE => { self.pos += 1; self.token = SyntaxKind::OpenBraceToken; return self.token; }
                // ... ~40 more punctuation arms: todo_port!("scan() punctuation, _tsc.js 9368 switch")
                _ => {
                    if is_identifier_start(ch) {
                        return self.scan_identifier(ch);
                    }
                    todo_port!("unicode whitespace / invalid character error 1127 arm");
                }
            }
        }
    }

    fn scan_identifier(&mut self, _first: i32) -> SyntaxKind {
        // fast ASCII loop, escape fallback (\u => scan_identifier_parts with
        // TokenFlags::UNICODE_ESCAPE), then the KEYWORD TABLE:
        // tsc textToKeywordObj (_tsc.js 8030). The table is GENERATED by
        // xtask codegen (keyword string -> SyntaxKind) — never hand-listed.
        todo_port!("scanIdentifier + textToKeywordObj lookup")
    }
}
```

```rust
/// returns the cooked text contribution; on invalid escape returns the
/// RAW slice (tsc: text.substring(start, pos)) and sets ContainsInvalidEscape
fn scan_extended_unicode_escape(&mut self, report: bool) -> String {
    let start = self.pos;          // at the backslash
    self.pos += 3;                 // consume `\u{`
    let escaped_start = self.pos;
    let value_str = self.scan_minimum_number_of_hex_digits(1, /*separators*/ false);
    let value: i64 = if value_str.is_empty() { -1 } else { i64::from_str_radix(&value_str, 16).unwrap_or(-1) };
    let mut invalid = false;
    if value < 0 {
        if report { self.error_here(&gen::Hexadecimal_digit_expected); }
        invalid = true;
    } else if value > 0x10FFFF {
        if report { self.error_at(escaped_start, self.pos - escaped_start,
            &gen::An_extended_Unicode_escape_value_must_be_between_0x0_and_0x10FFFF_inclusive); }
        invalid = true;
    }
    if self.pos >= self.end {
        if report { self.error_here(&gen::Unexpected_end_of_text); }
        invalid = true;
    } else if self.ch(self.pos) == c::CLOSE_BRACE {
        self.pos += 1;
    } else {
        if report { self.error_here(&gen::Unterminated_Unicode_escape_sequence); }
        invalid = true;
    }
    if invalid {
        self.token_flags |= TokenFlags::CONTAINS_INVALID_ESCAPE;
        return self.raw_slice(start, self.pos);
    }
    self.token_flags |= TokenFlags::EXTENDED_UNICODE_ESCAPE;
    utf16_encode_as_string(value as u32)
}
```

## [COPY] reScanGreaterToken (stage 1.6) — `_tsc.js` 9866

```rust
pub fn re_scan_greater_token(&mut self) -> SyntaxKind {
    if self.token == SyntaxKind::GreaterThanToken {
        let p = self.token_start + 1;   // after the first `>`
        if self.ch(p) == c::GREATER_THAN {
            if self.ch(p + 1) == c::GREATER_THAN {
                if self.ch(p + 2) == c::EQUALS {
                    self.pos = p + 3;
                    self.token = SyntaxKind::GreaterThanGreaterThanGreaterThanEqualsToken;
                    return self.token;
                }
                self.pos = p + 2;
                self.token = SyntaxKind::GreaterThanGreaterThanGreaterThanToken;
                return self.token;
            }
            if self.ch(p + 1) == c::EQUALS {
                self.pos = p + 2;
                self.token = SyntaxKind::GreaterThanGreaterThanEqualsToken;
                return self.token;
            }
            self.pos = p + 1;
            self.token = SyntaxKind::GreaterThanGreaterThanToken;
            return self.token;
        }
        if self.ch(p) == c::EQUALS {
            self.pos = p + 1;
            self.token = SyntaxKind::GreaterThanEqualsToken;
            return self.token;
        }
    }
    self.token
}
```

## [COPY] speculationHelper (stage 1.7) — `_tsc.js` 11099

```rust
impl<'t> Scanner<'t> {
    fn speculation_helper<R: Truthy>(&mut self, cb: impl FnOnce(&mut Self) -> R,
                                     is_lookahead: bool) -> R {
        let saved = self.save();
        let result = cb(self);
        // lookAhead ALWAYS rewinds; tryScan rewinds only on falsy result
        if is_lookahead || !result.is_truthy() { self.restore(saved); }
        result
    }
    pub fn look_ahead<R: Truthy>(&mut self, cb: impl FnOnce(&mut Self) -> R) -> R {
        self.speculation_helper(cb, true)
    }
    pub fn try_scan<R: Truthy>(&mut self, cb: impl FnOnce(&mut Self) -> R) -> R {
        self.speculation_helper(cb, false)
    }
}
pub trait Truthy { fn is_truthy(&self) -> bool; }
impl Truthy for bool { fn is_truthy(&self) -> bool { *self } }
impl<T> Truthy for Option<T> { fn is_truthy(&self) -> bool { self.is_some() } }
impl Truthy for SyntaxKind { fn is_truthy(&self) -> bool { *self != SyntaxKind::Unknown } }
```
